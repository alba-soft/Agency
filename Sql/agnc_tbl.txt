CREATE FUNCTION SEQ_NUM (PR_code IN VARCHAR2) RETURN NUMBER IS
PR_num     NUMBER;
BEGIN
   SELECT SQ_number
     INTO PR_num
     FROM sequences
    WHERE SQ_name = PR_code;

   UPDATE sequences
      SET SQ_number = SQ_number + 1
    WHERE SQ_name   = PR_code;

   RETURN( PR_num );
EXCEPTION
   WHEN NO_DATA_FOUND THEN
     PR_num := -1;
     RETURN( PR_num );
END;
